winstonflight

A winston wrapper to enable adding transports by config

Example

const transports = [{
    "type": "file",
    "options": {
        "level": "error",
        "filename": "error.log"
    }
}, {
    "type": "daily-rotate-file",
    "options": {
        "level": "info",
        "filename": "-all.log"
        "datePattern": "yyyy-MM-dd",
        "prepend": true
    }
}];

const logger = new (winston.Logger)({
    level: 'info',
    transports: winstonFlight(transports)
});
